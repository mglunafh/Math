\begin{defn} Пусть $f\in A[X],$ где $A$ --- коммутативное кольцо, $f = \sum\limits_{i=0}^n a_ix^i.$ Производная многочлена $f$ --- это многочлен $f' = \sum \limits_{i=1}^n ia_ix^{i-1}.$ 
\end{defn}

\begin{prop}[Свойства производной]
Пусть $f,g \in A[X], c\in A.$ Верны следующие утверждения:
\begin{enumerate}
	\item $(cf)' = cf'$;
	\item $(f+g)' =  f' +g'$;
	\item $(fg)' = f'g +fg'$;
	\item $(f^m)' = mf^{m-1}f'$ $\forall m\in \mathbb{N}.$
\end{enumerate}
\end{prop}
\begin{proof}
Итак:
\begin{enumerate}
	\item $$(cf)' = \left( \sum_{i=0}^n ca_ix^i \right)' = \sum _{i=1}^n ica_ix^{i-1} = c\sum _{i=1}^n ia_ix^{i-1} = cf'.$$
	\item Обозначим через $n= \max (\deg f, \deg g).$ 
	$$(f+g)' = \left(\sum_{i=0}^n (a_i+b_i)x^i\right)' = \sum _{i=1}^n i(a_i+b_i)x^{i-1} = \sum _{i=1}^n ia_ix^{i-1} + \sum _{i=1}^n ib_ix^{i-1} = f' + g'. $$ 
	\item Предыдущие два свойства означают, что операция взятия производной линейна, поэтому здесь достаточно установить, что свойство верно для произведения двух мономов:$ f = a_nx^n, g = b_m x^m.$ 
	
	$(fg)' = (n+m)a_nb_mx^{n+m-1} = 
	 na_nb_mx^{n+m-1} + ma_nb_mx^{n+m-1} =  na_nx^{n-1}b_mx^m + mb_mx^{m-1}a_nx^n = f'g + g'f.$ 
	 \item Докажем по индукции. База индукции классная: $f' = 1\cdot f' \cdot f^0.$
	 
	 Переход. Предположим, что для $m$ утверждение верно.
	 
	 $$(f^{m+1})' = (f^m\cdot f)' = (f^m)'f + f^m f' = mf^{m-1}f'f + f^m f' = (m+1)f^mf'.$$
\end{enumerate}
\end{proof}

\setcounter{cor}{0}
\begin{cor} $((x-a)^n)' = n(x-a)^{n-1}.$
\end{cor}

\begin{prop} Пусть $\k$ --- поле, $f\in \k[X], c\in \k$ --- корень $f$. $c$ --- кратный корень $f$, если и только если $f'(c) = 0.$ 
\end{prop}
\begin{proof}
$f' = ((x-c)g)' = g + (x-c)g'$. $f'(c) = g(c),$ поэтому если $c$ --- корень производной, то для $f$ $c$ --- кратный корень, и наоборот.
\end{proof}

\begin{cor} Пусть $f\in \k[X], c\in \k.$ $c$ --- кратный корень $f \Leftrightarrow$ $c$ --- корень $\gcd (f,f').$
\end{cor}

\begin{prop}
Пусть $\k$ --- поле характеристики $0$, $f\in \k[X],$ $c$ --- корень $f$ кратности $n\geqslant 1.$ В таком случае $c$ --- корень производной $f'$ степени $n-1$.
\end{prop}
\begin{proof} $f = (x-c)^ng, g(c) \neq 0.$
$$ f' = n(x-c)^{n-1}g + (x-c)^ng' = (x-c)^{n-1}(\underbrace{ng + (x-c)g'}_{h}),$$
$h(c) =n\cdot g(c) \neq 0$ в силу того, что у $\k$ характеристика нулевая и что попало, в частности, $n$ нулю не равняется. Отсюда $c$ --- корень кратности $n-1$.
\end{proof}

\begin{cor}
 $c\in A$ --- корень $f$ кратности $n$ в том и только в том случае, если $f(c)=f'(c)=\ldots = f^{(n-1)}(c) = 0,$ а $f^{(n)}(c) \neq 0.$
\end{cor}
\begin{proof} 
$\Leftarrow:$ Если $k \leqslant n,$ то $c$ --- корень $f^{(k)}$ степени $n-k$ следуя предыдущему предложению. В частности, $c$ --- простой корень $f^{(n-1)}$, поэтому $f^{(n)}(c)\neq 0.$

$\Rightarrow:$ Из того, что $n$ прозводных, начиния с нулевой, подряд обращаются в точке $c$ в нуль, следует, что кратность $c$ не меньше $n$. Тот факт, что $f^{(n)}$ уже не равна нулю в $c,$ означает, что $c$ является простым корнем для $f^{(n-1)}$ и корнем кратности ровно $n$ для $f.$ 
\end{proof}

\subsection{Формула Тейлора}

\begin{prop} Пусть $\k$ --- поле характеристики $0$, $f\in \k[X],$ $\deg f = n, c\in \k.$ Тогда верно следующее представление для $f:$
$$ f = \sum_{i=0}^n \frac{f^{(i)}(c)} {i!} (x-c)^i. $$
\end{prop}
\begin{proof} Сначала докажем (по индукции, естественно), что $f$ раскладывается в сумму степеней $(x-c)$ c какими-нибудь коэффициентами $a_i\in \k.$

База индукции --- это деление $f$ с остатком на $(x-c):$ $f = g(x-c) + a_0.$

Переход: предположим, что многочлен $g$ степени $n-1$ уже раскладывается в $\sum\limits_{i=0}^{n-1}b_ix^i,$ и тогда 
$$ f = \sum \limits_{i=0}^{n-1} b_i(x-c)^{i+1} + a_0 = \sum\limits_{i=0}^na_i(x-c)^i,$$ причем $a_i = b_{i-1}$ при $i\geqslant 1.$

Теперь заметим, что для степеней $(x-c)$ такое разложение единственно, ведь $1, (x-c), (x-c)^2, \ldots, (x-c)^n$ линейно независимы в $\k[X].$

$$ ((x-c)^i)^{(j)} = \left\{ \begin{matrix} 0 , && i < j; \\ i(i-1)\ldots (i-j+1)(x-c)^{i-j}, && i \geqslant j. \end{matrix} \right.  $$

Отсюда 
$$ f^{(j)} = \sum\limits_{i=0}^n a_i((x-i)^i)^{(j)} =  \sum\limits_{i=j}^na_ii(i-1)\ldots(i-j+1)(x-c)^{i-j},$$
$$ f^{(j)}(c) = a_j\cdot j\cdot(j-1)\cdot\ldots \cdot 1  = j! a_j,$$
$$ a_j = \frac{f^{(j)}(c)} {j!}.$$
На предпоследнее и последнее неравенство, в принципе, можно обратить внимание, потому что, чтобы факториал случайно не обнулился и не пришлось потом делить на нуль, требуется нулевая характеристика поля.
\end{proof}

\subsection{Многочлены над $\mathbb{R}$ и $\mathbb{C}$}

\begin{defn} Поле называется алгебраически замкнутым, если любой многочлен над этим полем степени, превышающей нулевую, имеет корень из поля.
\end{defn}

\begin{prop} Пусть $\k$ --- поле. Тогда $\k$ алгебраически замкнуто тогда и только тогда, когда все неприводимые многочлены с единичным старшим коэффициентом имеют вид  $x-c,\ \ c\in \k.$
\end{prop}
\begin{proof} 
$\Rightarrow:$ Рассмотрим неприводимый многочлен $f\in \k[X]$, $\deg f \geqslant 1.$ У него есть корень $c,$ и поэтому $f=g(x-c).$ Но поскольку $(x-c)$ необратим, а $f$ неприводим, обратимым будет $g:$ $g\in (\k[X])^* = \k \backslash \{0\}.$

$\Leftarrow:$ Многочлен $f$ степени, больше или равной $1$, делится на некоторый неприводимый, который в силу предположения имеет вид $\varepsilon (x-c),$ а тогда $f(c)=0.$
\end{proof}

\setcounter{cor}{0}
\begin{cor} В алгебраически замкнутом поле любой ненулевой многочлен $f$ представляется однозначно с точностью до порядка сомножителей в виде произведения 
$$ f = \varepsilon \prod\limits_{i=1}^l (x-c_i)^{n_i}, $$
где $l\geqslant 0,$ $\varepsilon \in \k^*,$ $c_1, \ldots, c_l$ --- различные элементы $\k,$ $n_i \geqslant 1.$ 
\end{cor}

\begin{cor}
В алгебраически замкнутом поле степень (ненулевого) многочлена равна сумме кратностей его корней.
\end{cor}

Следующая теорема идет без доказательства.
\begin{thm}[Основная теорема алгебры] Поле $\mathbb{C}$ алгебраически замкнуто. 
\end{thm}

В этом месте переходим к изучению ситуации с многочленами собственно над  $\mathbb{R}.$

Постараемся классифицировать неприводимые многочлены над $\mathbb{R}.$
\begin{itemize}
	\item Многочлены степени $1$, понятное дело, неприводимы.
	\item Если многочлен $f$ имеет степень $2$ и он приводим, то это значит, что $f = f_1 f_2,$ $\deg f_1 = \deg f_2 = 1,$ что тоже самое, что $\exists c\in \mathbb{R}: f(c) = 0.$ Ну а корень у крадратного трехчлена существует в том и только в том случае, когда его дискриминант неотрицателен. Итак, неприводимые многочлены над $\mathbb{R}$ степени $2$ --- это многочлены с отрицательным дискриминантом.
	\item Исходя из аналитических соображений можно показать, что любой многочлен третьей степени не является неприводимым, поскольку всегда имеет корень на вещественной оси.
\end{itemize}

На самом деле, все многочлены степеней выше второй неприводимыми не являются. Для начала докажем небольшую лемму.

\begin{lem} Пусть $f\in \mathbb{R}[X],$ $z \in \mathbb{C}.$ $f(z) = 0 \Leftrightarrow f(\overline{z}) = 0.$ 
\end{lem}
\begin{proof} Многочлен $f$ имеет вид $\sum\limits_{i=0}^n a_ix^i.$
$$0 =  f(z) = \sum\limits_{i=0}^n a_iz^i = \overline{\left(\sum\limits_{i=0}^n a_iz^i\right)}  = \sum\limits_{i=0}^n \overline{a_iz^i} = \sum\limits_{i=0}^n a_i(\overline{z})^i = f(\overline{z}).$$
\end{proof}
 
\begin{prop} Все неприводимые унитарные многочлены над $\mathbb{R}$ имеют вид:
\begin{enumerate}
	\item $x-c,$ $c\in \mathbb{R};$
	\item $x^2 +px+q,$ $p^2-4q < 0.$
\end{enumerate}
\end{prop}
\begin{proof}
Рассмотрим многочлен $f\in \mathbb{R}[X],$ $\deg f \geqslant 1.$ У $f$ как у многочлена над полем комплексных чисел имеются какие-то корни, и по основной теореме алгебры кратности корней в сумме дают $\deg f.$ Пусть $a_1,\ldots, a_l $ --- это вещественные корни, а $c_1, \overline{c_1}, \ldots, c_n, \overline{c_n}$ --- это все комплексные корни, идущие по вышедоказананной лемме парами. Докажем, что и кратности у сопряженных комплексных корней совпадают.

$f = (x-c_i)^mg,$ $g(c_i) \neq 0,$ $g\in \mathbb{C}.$ Применим к этому равенству операцию сопряжения --- это значит, что нужно пририсовать  знак сопряжения ко всем коэффициентам.

$$ f = \overline{f} = \overline{(x-c_i)^m g} = \overline{(x-c_i)^m} \overline{g} = (x-\overline{c_i})^m\overline{g},$$
причем $\overline{g}(\overline{c_i}) = \overline{g(c_i)} \neq 0,$ поэтому $m$ --- это кратность $\overline{c_i}$ как корня многочлена $\overline{f}=f.$

Итак,

$$ f = \varepsilon \prod_{i=1}^l (x-a_i)^{m_i} \prod_{i=1}^n (x-c_i)^{q_i} \prod_{i=1}^n (x-\overline{c_i})^{q_i} = 
 \varepsilon \prod_{i=1}^l (x-a_i)^{m_i} \prod_{i=1}^n ((x-c_i)(x-\overline{c_i}))^{q_i} = $$
 
 $$= \varepsilon \prod_{i=1}^l (x-a_i)^{m_i} \prod_{i=1}^n ( x^2 - 2 (\Re c_i)x + \left|c_i\right|^2 )^{q_i}, $$
 
причем $x^2 - 2 (\Re c_i)x + \left|c_i\right|^2 = f_{c_i} \in \mathbb{R}[X],$ и поскольку $\deg f_{c_i} = 2$ и уже имеются два мнимых корня, то никаких других корней у $f_{c_i}$ быть не может, поэтому $f_{c_i}$ не имеет вещественных корней.
\end{proof}
